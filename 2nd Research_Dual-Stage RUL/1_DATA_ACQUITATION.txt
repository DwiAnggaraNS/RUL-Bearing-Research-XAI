    "        first_chunk = True\n",
    "        for chunk in pd.read_csv(csv_file, dtype=COLUMN_DTYPES, chunksize=8192):\n",
    "            chunk.rename(columns=RION_TO_USED_FORMAT, inplace=True)\n",
    "\n",
    "            # Get the first time value in the chunk\n",
    "            first_time_value = chunk[\"time\"].iloc[0] if not chunk.empty else 0\n",
    "\n",
    "            # Substract time with first time value\n",
    "            chunk[\"time\"] = chunk[\"time\"] - first_time_value \\\n",
    "                + last_time_value # Adjust time based on the last value\n",
    "\n",
    "            # Update last_time_value for the next chunk\n",
    "            last_time_value = chunk[\"time\"].iloc[-1] if not chunk.empty else last_time_value\n",
    "\n",
    "            with open(output_file, 'a') as stream_out:\n",
    "                chunk.to_csv(stream_out, header=not header_written, index=False)\n",

    "            first_chunk = False\n",

    "def compute_fft(partition):\n",
    "    fft_results = []\n",
    "    \n",
    "    # Convert the partition to a NumPy array for efficient slicing\n",
    "    partition_array = partition.to_numpy()\n",
    "    \n",
    "    # Process the data in chunks of 2560 rows\n",
    "    for i in range(0, len(partition_array), SAMPLING_RATE):\n",
    "        chunk = partition_array[i:i + SAMPLING_RATE]\n",
    "        \n",
    "        # Skip if the chunk is incomplete\n",
    "        if len(chunk) < SAMPLING_RATE:\n",
    "            continue\n",
    "        \n",
    "        # Extract X, Y, Z columns (assuming they exist in the partition)\n",
    "        x = np.array(chunk[:, partition.columns.get_loc(\"X\")])\n",
    "        y = np.array(chunk[:, partition.columns.get_loc(\"Y\")])\n",
    "        z = np.array(chunk[:, partition.columns.get_loc(\"Z\")])\n",
    "        \n",
    "        # Compute FFT and frequency bins\n",
    "        fft_x = np.abs(rfft(x))\n",
    "        fft_y = np.abs(rfft(y))\n",
    "        fft_z = np.abs(rfft(z))\n",
    "        frequencies = rfftfreq(len(x), d=1 / SAMPLING_RATE)\n",
    "        \n",
    "        # Create a dictionary for this chunk with frequency bins as keys\n",
    "        fft_row = {f\"FFT_{int(freq)}Hz_X\": amp for freq, amp in zip(frequencies, fft_x)}\n",
    "        fft_row.update({f\"FFT_{int(freq)}Hz_Y\": amp for freq, amp in zip(frequencies, fft_y)})\n",
    "        fft_row.update({f\"FFT_{int(freq)}Hz_Z\": amp for freq, amp in zip(frequencies, fft_z)})\n",
    "        \n",
    "        # Include the time from column time \n",
    "        if 'time' in partition.columns:\n",
    "            time_value = chunk[0, partition.columns.get_loc(\"time\")]\n",
    "            fft_row['time'] = time_value\n",
    "        else:\n",
    "            raise ValueError(\"Column 'time' not found in the partition.\")\n",
    "        \n",
    "        fft_results.append(fft_row)\n",
    "    \n",
    "    # Convert the list of dictionaries to a DataFrame\n",
    "    return pd.DataFrame(fft_results)"
   ]
